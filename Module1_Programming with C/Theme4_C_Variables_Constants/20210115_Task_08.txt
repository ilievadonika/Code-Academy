const char c_chNEW_LINE = '\n';

	printf("Constant: c_uMAX_LOOP_COUNT = %d %c", c_uMAX_LOOP_COUNT, c_chNEW_LINE);
	printf("Constant: c_dfREAL_VALUE = %lf %c", c_dfREAL_VALUE, c_chNEW_LINE);

return 0;
}
